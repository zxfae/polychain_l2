8/// Transaction Sequencer for fair and deterministic ordering
9#[derive(Debug)]
10pub struct TransactionSequencer<C: CryptographyBridge> {
11    pending_transactions: Vec<PolyTransaction>,
12    sequence_counter: u64,
13    ordering_strategy: OrderingStrategy,
14    cryptography: C,
15    private_keys: HashMap<String, C::SecretKey>,
16}
17
18#[derive(Debug, Clone)]
19pub enum OrderingStrategy {
20    FirstComeFirstServed,
21    PriorityFee,
22    FairOrdering,
23    VrfRandom,
24}
25
26impl<C: CryptographyBridge> TransactionSequencer<C> {
27    pub fn new(cryptography: C, private_keys: HashMap<String, C::SecretKey>) -> Self {
28        Self {
29            pending_transactions: Vec::new(),
30            sequence_counter: 0,
31            ordering_strategy: OrderingStrategy::FairOrdering,
32            cryptography,
33            private_keys,
34        }
35    }
36
37    pub fn add_transaction(&mut self, transaction: PolyTransaction) -> Result<String, String> {
38        if !transaction.is_valid() {
39            return Err("Invalid transaction".to_string());
40        }
41
42        self.pending_transactions.push(transaction);
43        self.sequence_counter += 1;
44        Ok(format!("seq_tx_{}", self.sequence_counter))
45    }
46
47    pub fn sequence_batch(&mut self, batch_size: usize) -> Vec<PolyTransaction> {
48        let take_count = batch_size.min(self.pending_transactions.len());
49
50        match self.ordering_strategy {
51            OrderingStrategy::FirstComeFirstServed => {
52                // Simple FIFO
53                self.pending_transactions.drain(0..take_count).collect()
54            }
55            OrderingStrategy::FairOrdering => {
56                // Anti-MEV fair ordering using cryptographically secure hash
57                self.pending_transactions.sort_by_key(|tx| {
58                    let combined_data = format!("{}:{}", tx.sender, tx.time_stamp);
59                    let hash = blake3::hash(combined_data.as_bytes());
60                    // Convert first 8 bytes to u64 for sorting
61                    u64::from_le_bytes(hash.as_bytes()[0..8].try_into().unwrap_or([0; 8]))
62                });
63                self.pending_transactions.drain(0..take_count).collect()
64            }
65            _ => {
66                // Default to FIFO for other strategies
67                self.pending_transactions.drain(0..take_count).collect()
68            }
69        }
70    }
71
72    pub fn pending_count(&self) -> usize {
73        self.pending_transactions.len()
74    }
75
76    pub fn set_ordering_strategy(&mut self, strategy: OrderingStrategy) {
77        self.ordering_strategy = strategy;
78    }
79}

81/// PoS Consensus with VRF (ton code)
82// Removed std::time::Instant import for IC compatibility
83
84type ProposerResult<C> = Result<Vec<(String, Vec<u8>, C)>, CryptographyError>;
85
86#[derive(Debug)]
87pub struct AlgoConsensus<C: CryptographyBridge>
88where
89    C::SignedMessage: std::fmt::Debug + Clone + Sync,
90{
91    participation_threshold: f64,
92    max_proposers: usize,
93    validators: usize,
94    round_timeout: u64,
95    cryptography: C,
96    private_keys: HashMap<String, C::SecretKey>,
97}
98
99impl<C: CryptographyBridge> AlgoConsensus<C>
100where
101    C::SignedMessage: std::fmt::Debug + Clone + Sync,
102{
103    pub fn new(
104        participation_threshold: f64,
105        max_proposers: usize,
106        validators: usize,
107        round_timeout: u64,
108        cryptography: C,
109        private_keys: HashMap<String, C::SecretKey>,
110    ) -> Result<Self, CryptographyError> {
111        let (_pk, sk) = cryptography.key_generator()?;
112
113        // Test basique de signature au lieu de VRF
114        if cryptography.sign(&sk, b"test").is_err() {
115            return Err(CryptographyError::SigningError);
116        }
117        Ok(Self {
118            participation_threshold,
119            max_proposers,
120            validators,
121            round_timeout,
122            cryptography,
123            private_keys,
124        })
125    }
126
127    pub fn create_instance(
128        cryptography: C,
129        private_keys: HashMap<String, C::SecretKey>,
130    ) -> Result<Self, CryptographyError> {
131        Self::new(0.1, 5, 3, 5000, cryptography, private_keys)
132    }
133
134    fn create_seed(&self, prev_hash: &[u8; 32], time_stamp: i64) -> [u8; 32] {
135        use blake3::Hasher;
136        let mut hasher = Hasher::new();
137        hasher.update(prev_hash);
138        hasher.update(&time_stamp.to_le_bytes());
139        hasher.finalize().into()
140    }
141
142    fn select_proposers(
143        &self,
144        balances: &HashMap<String, f64>,
145        seed: &[u8; 32],
146        _public_keys: &HashMap<String, C::PublicKey>,
147    ) -> ProposerResult<C::SignedMessage> {
148        let total_stake: f64 = balances.values().sum();
149        if total_stake <= 0.0 {
150            return Err(CryptographyError::SigningError);
151        }
152
153        let mut candidates: Vec<(String, f64, Vec<u8>, C::SignedMessage)> = Vec::new();
154
155        for (address, stake) in balances.iter().filter(|(_, balance)| **balance > 0.0) {
156            let weight = *stake / total_stake;
157            if weight < self.participation_threshold {
158                continue;
159            }
160
161            if let Some(private_key) = self.private_keys.get(address) {
162                if let Ok((vrf_output, proof)) = self.vrf_generate_internal(private_key, seed) {
163                    let priority = self.vrf_output(&vrf_output) * weight;
164                    candidates.push((address.clone(), priority, vrf_output.to_vec(), proof));
165                }
166            }
167        }
168
169        candidates.sort_by(|a, b| b.1.partial_cmp(&a.1).unwrap_or(std::cmp::Ordering::Equal));
170
171        let selected = candidates
172            .into_iter()
173            .take(self.max_proposers)
174            .map(|(addr, _, output, proof)| (addr, output, proof))
175            .collect();
176
177        Ok(selected)
178    }
179
180    fn vrf_output(&self, output: &[u8]) -> f64 {
181        use blake3::Hasher;
182        let mut hasher = Hasher::new();
183        hasher.update(output);
184        let hash = hasher.finalize();
185        let first_bytes = &hash.as_bytes()[0..8];
186        u64::from_le_bytes(first_bytes.try_into().unwrap()) as f64 / u64::MAX as f64
187    }
188
189    pub fn run_consensus_round(
190        &self,
191        prev_hash: &[u8; 32],
192        transactions: Vec<PolyTransaction>,
193        balances: &HashMap<String, f64>,
194        public_keys: &HashMap<String, C::PublicKey>,
195    ) -> Result<PolyBlock, CryptographyError> {
196        let start_time = ic_cdk::api::time();
197
198        // Convert hash for compatibility
199        let prev_hash_string = hex::encode(prev_hash);
200        let block = PolyBlock::new(transactions, prev_hash_string);
201        let seed = self.create_seed(prev_hash, block.timestamp);
202
203        let proposers = self.select_proposers(balances, &seed, public_keys)?;
204        if proposers.is_empty() {
205            return Err(CryptographyError::SigningError);
206        }
207
208        // Simplified consensus check
209        let is_approved = proposers.len() >= self.validators.min(2);
210        if !is_approved {
211            return Err(CryptographyError::SigningError);
212        }
213
214        // Check timeout using IC time (nanoseconds)
215        let elapsed_ns = ic_cdk::api::time() - start_time;
216        let elapsed_ms = elapsed_ns / 1_000_000; // Convert to milliseconds
217        if elapsed_ms > self.round_timeout {
218            return Err(CryptographyError::SigningError);
219        }
220
221        Ok(block)
222    }
223}

225// Extension VRF pour le consensus
226impl<C: CryptographyBridge> AlgoConsensus<C>
227where
228    C::SignedMessage: std::fmt::Debug + Clone + Sync,
229{
230    // Méthode VRF intégrée qui utilise les méthodes crypto existantes
231    fn vrf_generate_internal(
232        &self,
233        secret_key: &C::SecretKey,
234        seed: &[u8],
235    ) -> Result<(Vec<u8>, C::SignedMessage), CryptographyError> {
236        // Fallback: utilise signature normale comme VRF
237        let signature = self.cryptography.sign(secret_key, seed)?;
238        Ok((seed.to_vec(), signature))
239    }
