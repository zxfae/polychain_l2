35/// ML-DSA-44 Post-Quantum signature scheme
36/// Pure Rust implementation for WASM compatibility
37/// # Features
38/// - Key generation with randomness
39/// - Message signing/verification via ML-DSA-44
40/// - Cross-Compatibility through CryptographyBridge trait
41#[derive(Debug, Clone)]
42pub struct Mldsa44;
43
44impl CryptographyBridge for Mldsa44 {
45    type PublicKey = MLDSAPublicKey;
46    type SecretKey = PrivateKey;
47    type SignedMessage = MLDSASignature;
48
49    fn key_generator(&self) -> Result<(Self::PublicKey, Self::SecretKey), CryptographyError> {
50        let mut rng = OsRng;
51        let (private_key, public_key) = ml_dsa_44::try_keygen_with_rng(&mut rng)
52            .map_err(|_| CryptographyError::KeyGeneration)?;
53        Ok((MLDSAPublicKey(private_key), public_key))
54    }
55
56    fn sign(
57        &self,
58        secret_key: &Self::SecretKey,
59        message: &[u8],
60    ) -> Result<Self::SignedMessage, CryptographyError> {
61        let mut rng = OsRng;
62        let signature = secret_key
63            .try_sign_with_rng(&mut rng, message, &[])
64            .map_err(|_| CryptographyError::SigningError)?;
65        Ok(MLDSASignature(signature.to_vec()))
66    }
67
68    fn verify(
69        &self,
70        public_key: &Self::PublicKey,
71        message: &[u8],
72        signature: &Self::SignedMessage,
73    ) -> Result<bool, CryptographyError> {
74        // Validate signature length first - critical for security
75        if signature.0.len() != 2420 {
76            return Err(CryptographyError::SigningError);
77        }
78        
79        let sig_array: [u8; 2420] = signature.0.clone().try_into()
80            .map_err(|_| CryptographyError::SigningError)?;
81        let is_valid = public_key.0.verify(message, &sig_array, &[]);
82        Ok(is_valid)
83    }
84
85    fn public_key_to_bytes(&self, public_key: &Self::PublicKey) -> Vec<u8> {
86        public_key.0.clone().into_bytes().to_vec()
87    }
88}

100    #[test]
101    fn mldsa44key_generator() {
102        start_log();
103        info!("Testing key generator for ML-DSA-44");
104        let cryptography = Mldsa44;
105
106        let generation = cryptography.key_generator();
107        assert!(generation.is_ok(), "Failed to generate ML-DSA-44 keypair");
108
109        if let Ok((public_key, _secret_key)) = generation {
110            debug!("Secret key generated successfully");
111            let public_key_bytes = cryptography.public_key_to_bytes(&public_key);
112            debug!("Public key length : {:?}", public_key_bytes.len());
113            let pkl = public_key_bytes.len();
114            assert!(pkl > 0, "Public key should not be empty");
115            info!("Key generation successful with correct key size");
116        }
117    }

119    #[test]
120    fn mldsa44_sign_verify() {
121        start_log();
122        info!("Testing sign and verify for ML-DSA-44");
123        let cryptography = Mldsa44;
124
125        match cryptography.key_generator() {
126            Ok((public_key, secret_key)) => {
127                let message = b"Test message for signing";
128                match cryptography.sign(&secret_key, message) {

131
